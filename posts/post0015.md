<!-- 3. 스크립트 (휠 스크롤 및 팝업 제어) -->
<script>
  let currentScale = 1;
  const minScale = 0.5;
  const maxScale = 4;
  const step = 0.1;

  function openPopup() {
    currentScale = 1;
    const img = document.getElementById('popupImg');
    if (img) img.style.transform = `scale(${currentScale})`;
    document.getElementById('imgPopup').showModal();
  }

  function closePopup() {
    document.getElementById('imgPopup').close();
  }

  // 마우스 휠 스크롤 확대/축소 이벤트 리스너
  document.getElementById('imgPopup').addEventListener('wheel', function(e) {
    e.preventDefault(); // 페이지 스크롤 방지
    
    const img = document.getElementById('popupImg');
    if (!img) return;

    if (e.deltaY < 0) {
      // 휠 올림 -> 확대
      currentScale = Math.min(maxScale, currentScale + step);
    } else {
      // 휠 내림 -> 축소
      currentScale = Math.max(minScale, currentScale - step);
    }

    img.style.transform = `scale(${currentScale})`;
  }, { passive: false });
</script>
