파이썬 코드 설명

$A = \pi r^2$

import math

def calculate_circle_area(radius):
    """원의 넓이를 계산하여 반환합니다."""
    if radius < 0:
        raise ValueError("반지름은 음수일 수 없습니다.")
        
    return math.pi * (radius ** 2)

# 코드 실행 예시
print(calculate_circle_area(5.0))
